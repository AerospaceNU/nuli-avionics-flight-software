6/**
7 * @class Debounce
8 * @brief Utility class to debounce boolean signals.
9 * @details
10 * This class provides a mechanism to debounce signals, typically used to prevent
11 * spurious transitions in digital inputs due to noise or mechanical switch bounce.
12 * It ensures that a condition must remain true for a specified amount of time
13 * before it is considered valid.
14 */
15class Debounce {
16public:
17 /**
18 * @brief Construct a new Debounce object with a specified debounce time.
19 * @param debounceTimeMs The debounce time in milliseconds. The condition must
20 * remain true for at least this duration before being
21 * considered valid.
22 */
23 explicit Debounce(const uint32_t debounceTimeMs) : m_debounceTimeMs(debounceTimeMs) {}

32 * @brief Set the debounce time.
33 * @param debounceTimeMs The new debounce time in milliseconds.
34 */
35 void setDebounceTime(const uint32_t debounceTimeMs) {
36 m_debounceTimeMs = debounceTimeMs;
37 }
38
39 /**
40 * @brief Check if a condition has been stable for the debounce time.
41 * @details When called with a boolean condition and the current timestamp, this
42 * function returns true only if the condition has been continuously true
43 * for at least the debounce time specified during construction or via
44 * setDebounceTime(). If the condition is false, the internal reference
45 * time is reset.
46 * @param condition The boolean condition to debounce.
47 * @param timestamp The current timestamp as a Timestamp_s object, where runtime_ms represents the elapsed time in milliseconds.
48 * @return true if the condition has been true for the debounce duration, false otherwise.
49 */
50 bool check(const bool condition, const Timestamp_s& timestamp) {
51 if (!m_isInitialized) {
52 m_referenceTime = timestamp.runtime_ms;
53 m_isInitialized = true;
54 }
55 if (condition) {
56 if (timestamp.runtime_ms - m_referenceTime >= m_debounceTimeMs) {
57 return true;
58 }
59 } else {
60 m_referenceTime = timestamp.runtime_ms;
61 }
62 return false;
63 }
64
65 /**
66 * @brief Reset the debounce state.
67 * @details This function clears the internal state, including the reference time and
68 * initialization flag. After calling reset(), the next call to check() will
69 * treat the condition as if it were the first time being checked.
70 */
71 void reset() {
72 m_isInitialized = false;
73 }
74
75private:
76 uint32_t m_debounceTimeMs; ///< Time in milliseconds for which the condition must remain true to be considered valid.
77 uint32_t m_referenceTime = 0; ///< Reference timestamp used for debouncing.
78 bool m_isInitialized = false; ///< Flag indicating whether the debounce object has been initialized.
79};
