26 numBeeps = 1;
27 } else {
28 numBeeps = 0;

32

35 constexpr uint32_t beepOnMs = 100; // short beep
36 constexpr uint32_t beepOffMs = 100; // gap between beeps
37 constexpr uint32_t cycleGapMs = 800; // gap after full sequence
38
39 const uint32_t sequenceDuration = numBeeps * (beepOnMs + beepOffMs) + cycleGapMs;
40 const uint32_t t = timestamp.runtime_ms % sequenceDuration;
41
42 for (int b = 0; b < numBeeps; b++) {
43 const uint32_t start = b * (beepOnMs + beepOffMs);
44 const uint32_t end = start + beepOnMs;
45 if (t >= start && t < end) {
46 active = true;
47 break;
48 }
49 }
50 } else {
51 // === No continuity: long beep, short pause ===
52 constexpr uint32_t longBeepMs = 1000;
53 constexpr uint32_t shortPauseMs = 2000;
54 constexpr uint32_t sequenceDuration = longBeepMs + shortPauseMs;
55 const uint32_t t = timestamp.runtime_ms % sequenceDuration;
56
57 if (t < longBeepMs) {
58 active = true;
59 } else {
60 active = false;
61 }
62 }
