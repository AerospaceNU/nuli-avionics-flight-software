22/**
23 * @class ConfigurationData
24 * @brief Wrapper object for users to interact with configuration variables
25 * @details Tracks updates, allowing for automatic writes to FRAM with Configuration.pushUpdatesToMemory();
26 * @tparam T Data type of the configuration varable. Must match ConfigurationRegistry.h
27 */
28template <typename T>
29class ConfigurationData {
30public:
31 /**
32 * @brief Default constructor
33 * @details Used to allow temporary invalid objects to be made in classes
34 * before the varable is assigned to a valid object with Configuration.getConfigurable<>();
35 */
36 ConfigurationData() = default;
37
38 /**
39 * @brief Constructs a new ConfigurationData, used internally to Configuration
40 * @details This class acts largely as a pointer container, letting Configuration
41 * data to be promoted to the templated type.
42 * @param b BaseConfigurationData_s that this object will reference
43 */
44 explicit ConfigurationData(BaseConfigurationData_s* b) : base(b) {}
45
46 /**
47 * @brief Getter method for the configuration variable
48 * @details Will return default constructed T if the default constructor was used.
49 * @return The value of the configuration variable
50 */
51 T get() const {
52 if (!base || !base->data) return T{}; // default-constructed T
53 return *reinterpret_cast<const T*>(base->data);
54 }
55
56 /**
57 * @brief Sets the value of the configuration varable
58 * @details Does nothing if object was created with the default constructor
59 * @param newVal Value to set the configuration variable to
60 */
61 void set(const T& newVal) {
62 if (!base || !base->data) return;
63 *reinterpret_cast<T*>(base->data) = newVal;
64 base->m_isUpdated = true;
65 }
66
67 /**
68 * @brief Determines if the object is a valid configuration varable
69 * @details If the Configuration object was not created with your
70 * intended configuration varable, or if the default constructor was
71 * used this will return false.
72 * @return if the varable is valid
73 */
74 bool isValid() const { return base != nullptr; }
75
76private:
77 BaseConfigurationData_s* base = nullptr; ///< pointer to the underlying data
78};
