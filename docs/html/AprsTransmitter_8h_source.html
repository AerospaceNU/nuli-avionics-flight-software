18/**
19 * @class AprsTransmitter
20 * @brief Turns binary data into sin waves, output-ed on the DAC
21 * @details To modulate the signal, precomputed 1200 hz and 2200 hz sin tables were computed. 2200 represents a one and 1200 represents a zero.
22 * To provide a smooth transition between the two frequency's, you must start the next frequency at the same value and direction as the current one.
23 * This class handles computing where to start in each sin table for each bit, and then uses timers/interrupts to accurately modulate the DAC/
24 */
25class AprsTransmitter {
26public:
27 /**
28 * @brief Initializes the hardware timers needed for modulating the signal accurately. Must be called only once, before any other methods are called.
29 */
30 static void configure();
31
32 /**
33 * @brief Transmits binary data through the DAC
34 * @param data Binary data to transmit
35 * @param length How many bytes to transmit
36 */
37 static void send(const uint8_t* data, int32_t length);
38
39 /**
40 * @brief Begins a transition of the last data sent
41 */
42 static void beginTransmission();
43
44 /**

46 * @return if the signal modulation is active

48 static bool isCurrentlyTransmitting();

53 static constexpr uint32_t MAXIMUM_BITS = 1000; ///< Maximum message length in bits
54
55 static volatile bool m_transmitActive; ///< Tracks if the IRS is active
56 // The current bit's waveform

58 static uint32_t m_currentSampleSize; ///< Length of the current sample (should always be SAMPLE_FREQUENCY / BOD_RATE)
59 static uint16_t currentSampleValue; ///< @todo is this needed to be a member? Can probably be local or deleted
60 static uint32_t sampleIndex; ///< Index within the sin table

63 static uint32_t m_transmissionBitWaveformsNum; ///< How many bits we have to send
64 static BitSinTable_s m_transmissionBitWaveforms[MAXIMUM_BITS]; ///< Datastructure containing the sin table windows for every bit we will transmit
65};
