6 * @brief Utility class to debounce boolean signals.
7 * @details
8 * This class provides a mechanism to debounce signals, typically used to prevent

10 * It ensures that a condition must remain true for a specified amount of time
11 * before it is considered valid.

37 /**
38 * @brief Check if a condition has been stable for the debounce time.
39 * @details When called with a boolean condition and the current timestamp, this
40 * function returns true only if the condition has been continuously true
41 * for at least the debounce time specified during construction or via
42 * setDebounceTime(). If the condition is false, the internal reference
43 * time is reset.
44 * @param condition The boolean condition to debounce.
45 * @param runtime_ms The current timestamp, where runtime_ms represents the system elapsed time in milliseconds
46 * @return true if the condition has been true for the debounce duration, false otherwise.
47 */
48 bool check(const bool condition, const uint32_t runtime_ms) {
49 if (!m_isInitialized) {
50 m_referenceTime = runtime_ms;
51 m_isInitialized = true;
52 }
53 if (condition) {
54 if (runtime_ms - m_referenceTime >= m_debounceTimeMs) {
55 return true;
56 }
57 } else {
58 m_referenceTime = runtime_ms;
59 }
60 return false;
61 }
62
63 /**
64 * @brief Reset the debounce state.
65 * @details This function clears the internal state, including the reference time and
66 * initialization flag. After calling reset(), the next call to check() will
67 * treat the condition as if it were the first time being checked.
68 */
69 void reset() {
70 m_isInitialized = false;
71 }
72
73private:
74 uint32_t m_debounceTimeMs = 0; ///< Time in milliseconds for which the condition must remain true to be considered valid.
75 uint32_t m_referenceTime = 0; ///< Reference timestamp used for debouncing.
76 bool m_isInitialized = false; ///< Flag indicating whether the debounce object has been initialized.
77};
