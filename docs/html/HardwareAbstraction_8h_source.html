17/**
18 * This macro generates three methods for a given type of hardware:
19 * - addHardware(Hardware*);
20 * - getNumHardware();
21 * - getHardwareArray();
22 *
23 * ## works as string concatenation in macros, allowing for the type name to be injected into the method names
24 */
25#define GENERATE_GET_ADD_METHODS_MACRO(Type, arrayVariable, numVariable, MaxNum) \
26void add##Type(Type* instance) { \
27 if(numVariable >= MaxNum) return; \
28 arrayVariable[numVariable] = instance; \
29 numVariable++; \
30} \
31inline uint8_t getNum##Type##s() const { \
32 return numVariable; \
33} \
34inline Type* get##Type##Array() { \
35 return arrayVariable[0]; \
36}
