21/**
22 * This macro generates three methods for a given type of hardware:
23 * - addHardware(Hardware*);
24 * - getNumHardware();
25 * - getHardwareArray();
26 *
27 * ## works as string concatenation in macros, allowing for the type name to be injected into the method names
28 */
29#define GENERATE_GET_ADD_METHODS_MACRO(Type, arrayVariable, numVariable, MaxNum) \
30int16_t append##Type(Type* instance) { \
31 if(numVariable >= MaxNum) return -1; \
32 arrayVariable[numVariable] = instance; \
33 numVariable++; \
34 return numVariable - 1; \
35} \
36inline uint8_t getNum##Type##s() const { \
37 return numVariable; \
38} \
39inline Type* get##Type(uint8_t index) { \
40 return arrayVariable[index]; \
41}
42
43#define GENERATE_GET_SET_METHODS_MACRO(Type, arrayVariable) \
44void set##Type(Type* instance) { \
45 arrayVariable = instance; \
46} \
47inline Type* get##Type() { \
48 return arrayVariable; \
49}
