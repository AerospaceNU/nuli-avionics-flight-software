8/**
9 * @class ArduinoSerialReader
10 * @brief Reads newline-terminated text lines from the Arduino serial interface into a fixed-size buffer.
11 * @details This class provides a simple line reader implementation that reads characters from the global
12 * Arduino Serial interface. Characters are stored in an internal buffer up to a fixed capacity N.
13 * When a newline character ('\n') is received, the buffer is null-terminated and the function returns true.
14 * Optionally, received characters can be echoed back to the serial output as they are read.
15 * @tparam N The maximum size of the line buffer, including the null terminator.
16 */
17template <unsigned N>
18class ArduinoSerialReader final : public LineReader {
19public:
20 /**
21 * @brief Constructs a new ArduinoSerialReader.
22 * @details Initializes the internal buffer and configures whether to echo received characters.
23 * @param echo If each received character will be echoed back to the serial output.
24 */
25 explicit ArduinoSerialReader(const bool echo = false) : m_echo(echo) {}
26
27 /**
28 * @brief Reads characters from the serial buffer until a full line is received.
29 * @details This function reads all available characters from Serial. Characters are appended to an internal
30 * buffer until either the buffer limit N - ` is reached or a newline character ('\n') is received.
31 * When a newline is encountered, the buffer is null-terminated, the read index is reset,
32 * and the function returns `true` to indicate that a full line is ready.
33 * If echo mode is enabled, each received character is also immediately sent back via Serial.print()
34 * @return if a complete line ending in `'\n'` was read and is ready to be retrieved with getLine()
35 */
36 bool readLine() override {
37 while (Serial.available() > 0) {
38 const char c = Serial.read();
39 if (m_serialReadIndex < N - 1) {
40 m_serialRead[m_serialReadIndex++] = c;
41 if (m_echo) {
42 Serial.print(c);
43 }
44 }
45 if (c == '\n') {
46 m_serialRead[m_serialReadIndex] = '\0';
47 m_serialReadIndex = 0;
48 return true;
49 }
50 }
51 return false;
52 }
53
54 /**
55 * @brief Returns a pointer to the internal line buffer.
56 * @details The returned buffer contains the most recently read line, null-terminated.
57 * The buffer is overwritten on the next successful call to `readLine()`.
58 * @return A pointer to the internal character buffer containing the last line.
59 */
60 char* getLine() override {
61 return m_serialRead;
62 }
63
64private:
65 bool m_echo = false; ///< Whether to echo received characters back to the serial output.
66 char m_serialRead[N] = ""; ///< Internal buffer used to store the current line being read.
67 uint32_t m_serialReadIndex = 0; ///< Current write index within the line buffer.
68};
