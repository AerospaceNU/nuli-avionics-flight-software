70 // this is also an approximation
71
72 const float saturationVapourPressure = 6.1094f * exp((17.625f * (temperatureK - Units::C_TO_K)) / ((temperatureK - Units::C_TO_K) + 243.04f)); // in hPa
73 // equation can be seen from here (re-arrange RH) http://www.atmo.arizona.edu/students/courselinks/fall12/atmo336/lectures/sec1/humidity.html
74 const float vapourPressure = (humidityPercent / 100) * saturationVapourPressure;
