7/**
8 * @class DiscreteRotation
9 * @brief Represents a 3D rotation composed only of 90° increments around the X, Y, or Z axes.
10 * @details
11 * This class allows for efficient manipulation of axis-aligned rotations by storing
12 * the mapping of input axes to output axes and their corresponding signs. It supports
13 * composing rotations, inverting them, and applying the transformation to 3D vectors.
14 */
15class DiscreteRotation final : public Vector3DTransform {
16public:
17 /**
18 * @brief Constructs an identity rotation.
19 * @details
20 * Initializes the rotation such that X→X, Y→Y, and Z→Z, with all positive directions.
21 */
22 DiscreteRotation() : axis{0, 1, 2}, sign{1, 1, 1} {}

24 /**
25 * @brief Returns an identity rotation.
26 * @details
27 * Equivalent to the default constructor. The rotation does not alter any input vector.
28 * @return A DiscreteRotation representing no rotation.

30 static DiscreteRotation identity() {
31 return DiscreteRotation{};

34 /**
35 * @brief Applies this rotation to a 3D vector.
36 * @details
37 * The input vector's components are permuted and sign-flipped according to the rotation definition.
38 * @param v Input vector to transform.
39 * @return The transformed vector after applying the rotation.
40 */
41 Vector3D_s transform(const Vector3D_s& v) const override {
42 const float in[3] = {v.x, v.y, v.z};
43 return {
44 float(sign[0]) * in[axis[0]],
45 float(sign[1]) * in[axis[1]],
46 float(sign[2]) * in[axis[2]]
47 };
48 }

116 /**
117 * @brief Returns the inverse of this rotation.
118 * @details
119 * The inverse rotation reverses the effect of the current rotation, such that
120 * applying both consecutively yields the identity rotation.
121 * @return A DiscreteRotation representing the inverse rotation.
122 */
123 DiscreteRotation inverse() const {
124 DiscreteRotation inv{};
125 for (int i = 0; i < 3; ++i) {
126 // find which input axis maps to output i
127 for (int j = 0; j < 3; ++j) {
128 if (axis[j] == i) {
129 inv.axis[i] = j;
130 inv.sign[i] = sign[j];
131 }
132 }
133 }
134 return inv;
135 }

138 uint8_t axis[3]; ///< Axis mapping: 0 = X, 1 = Y, 2 = Z.
139 int8_t sign[3]; ///< Sign mapping: +1 or -1 for each axis direction.

143 * @details
144 * Used internally to define specific 90° rotations around axes.
145 * @param ax0 Output axis index for input X.
146 * @param ax1 Output axis index for input Y.
147 * @param ax2 Output axis index for input Z.
148 * @param s0 Sign for output X component.
149 * @param s1 Sign for output Y component.
150 * @param s2 Sign for output Z component.
151 */
152 DiscreteRotation(const uint8_t ax0, const uint8_t ax1, const uint8_t ax2,
153 const int8_t s0, const int8_t s1, const int8_t s2)
154 : axis{ax0, ax1, ax2}, sign{s0, s1, s2} {}
155
156 /**
157 * @brief Composes this rotation with another.
158 * @details
159 * Returns the result of applying the given rotation `other` after this one,
160 * effectively combining the two transformations in sequence.
161 * @param other The rotation to apply after this rotation.
162 * @return A new DiscreteRotation representing the composed rotation.
163 */
164 DiscreteRotation compose(const DiscreteRotation& other) const {
165 DiscreteRotation r{};
166 for (int i = 0; i < 3; ++i) {
167 r.axis[i] = other.axis[axis[i]];
168 r.sign[i] = sign[i] * other.sign[axis[i]];
169 }
170 return r;
171 }
