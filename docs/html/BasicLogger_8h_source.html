27 m_hardware = hardware;

34
35 // Setup CLI interface
36 m_offloadBinaryFlag.setDependency(&m_offloadBinaryFlag);

41
42 const uint32_t numEntries = m_flash->getMemorySizeBytes() / sizeof(InternalStruct_s);
43
44 uint32_t left = 0;
45 uint32_t right = numEntries; // exclusive upper bound
46 uint32_t firstEmpty = numEntries; // default if nothing is 0xFF
47
48 while (left < right) {
49 uint32_t mid = left + (right - left) / 2;
50 uint8_t id;
51 offload(mid, id);
52
53 if (id == 0xFF) {
54 firstEmpty = mid; // possible candidate
55 right = mid; // search left half
56 } else {
57 left = mid + 1; // search right half
58 }
59 }
60
61 // firstEmpty is now the first index where id == 0xFF
62 m_logWriteIndex = firstEmpty;
63
64 m_debug->message("Logging setup, starting at entry: %d, index: %d", getEntryNumber(), (m_logWriteIndex * sizeof(InternalStruct_s)));
65 m_debug->message("Maximum log length (s): %d, Remaining log length (s): %d", getMaxLogLengthSeconds(), getRemainingLogLengthSeconds());
66 logMessage("Logger setup");
67 }

139 void offloadCallback() {
140 uint32_t failCount = 0;
141 m_debug->message("Starting Offload");
142 m_debug->data(m_headerStr);
143 for (uint32_t i = 0; true; i++) {
144 uint8_t id;
145 const LogDataStruct logData = offload(i, id);
146 if (id == 0xFF) {
147 failCount++;
148 if (failCount >= 4) {
149 break;
150 }
151 } else if (id == 0x01) {
152 m_printFunction(logData, m_debug);
153 } else if (id == 0x02) {
154 m_debug->data("New flight");
155 } else if (id == 0x03) {
156 char* str = (char*)&logData;
157 str[sizeof(LogDataStruct) - 1] = '\0';
158 m_debug->data("%s", str);
159 }
160 }
161 m_debug->message("Ending Offload");
162 }
